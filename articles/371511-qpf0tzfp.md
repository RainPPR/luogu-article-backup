考前突击了一下 [流运算符（<<+>>）的重载](https://zh.cppreference.com/w/cpp/language/operators)，感觉会了的样子。

```cpp
template<typename tp>
istream& operator >>(istream &in, vector<tp> &a) {
    for (tp &t : a) in >> t;
    return in;
}

template<typename tp>
ostream& operator <<(ostream &out, vector<tp> &a) {
    out << "{"; if (!a.empty()) {
        out << a[0];
        for (size_t i = 1; i < a.size(); ++i)
        out << ", " << a[i];
    } return (out << "}", out);
}
```
